    \section*{Layer L ($1 \leq L \leq L_0$): }
    Input :
    \[A^{(L-1)}\]
    Weight : 
    \[W^{(L)} \ (m_L \times n_L)\]
    Bias : 
    \[B^{(L)}\]
    \[
        Z^{(L)} = W^{(L)} \cdot A^{(L-1)} + B^{(L)} \rightarrow
        Z^{(L)}_i = \sum_{j = 1}^{n_L}W^{(L)}_{i, j}A^{(L-1)}_j + B^{(L)}_j
    \]
    Output : 
    \[
        A^{(L)} = f_L(Z^{(L)}) \rightarrow 
        A^{(L)}_i = f_L(Z^{(L)}_i)
    \] 


    \section*{Loss :}
    Loss function :
    \[
        C = \frac{1}{2}\sum_{i = 1}^{n_{(L_0)}}\left(
            A^{(L_0)}_i - Y_i
        \right)^2
    \]
    \section*{Gradient Layer $L_0$ :}
    We have the partial derivative of $C$ respect to $W^{(L_0)}_{i, j}$ : 
    \[
        \frac{\partial C}{\partial W^{(L_0)}_{i, j}} = 
        \frac{\partial C}{\partial A^{(L_0)}_{i}} \cdot
        \frac{\partial A^{(L_0)}_{i}}{\partial Z^{(L_0)}_i} \cdot
        \frac{\partial Z^{(L_0)}_i}{\partial W^{(L_0)}_{i, j}}
    \]  
    Let
    \[
        \delta^{(L_0)}_i = 
        \frac{\partial C}{\partial A^{(L_0)}_{i}} \cdot
        \frac{\partial A^{(L_0)}_{i}}{\partial Z^{(L_0)}_i} =
        (A^{(L_0)}_i - Y_i) \cdot f^{\prime}_{L_0}\left(Z^{(L_0)}_i\right)
    \]
    And 
    \[
        \frac{\partial Z^{(L_0)}_i}{\partial W^{(L_0)}_{i, j}} = A^{(L_0-1)}_j
    \]  

    \[
        \Rightarrow \frac{\partial C}{\partial W^{(L_0)}_{i, j}} = 
        \delta^{(L_0)}_i \cdot A^{(L_0-1)}_j
    \]
    We have the partial derivative of $C$ respect to $B^{(L_0)}_i$ :
    \[
        \frac{\partial C}{\partial B^{(L_0)}_{i}} = 
        \frac{\partial C}{\partial A^{(L_0)}_{i}} \cdot
        \frac{\partial A^{(L_0)}_{i}}{\partial Z^{(L_0)}_i} \cdot
        \frac{\partial Z^{(L_0)}_i}{\partial B^{(L_0)}_{i}} = 
        \delta^{(L_0)}_i
    \]

    \section*{Gradient Layer $L_0 - 1$ :}
    We have the partial derivative of $C$ respect to $W^{(L_0 - 1)}_{i, j}$ : 
    \[
        \frac{\partial C}{\partial W^{(L_0-1)}_{i, j}} = 
        \sum_{k = 1}^{m_{L_0}} \left( 
            \frac{\partial C}{\partial A^{(L_0)}_k} \cdot
            \frac{\partial A^{(L_0)}_k}{\partial Z^{(L_0)}_k} \cdot
            \frac{\partial Z^{(L_0)}_k}{\partial A^{(L_0 - 1)}_i}
        \right) \cdot
        \left(
            \frac{\partial A^{(L_0 - 1)}_i}{\partial Z^{(L_0 - 1)}_i} \cdot
             \frac{\partial Z^{(L_0 - 1)}_i}{\partial W^{(L_0 - 1)}_{i, j}}
        \right)
    \]  
    \\
    As we defined above:
    \[
        \delta^{(L_0)}_i = 
        (A^{(L_0)}_i - Y_i) \cdot \frac{\partial A^{(L_0)}_{i}}{\partial Z^{(L_0)}_i} = 
        \frac{\partial C}{\partial A^{(L_0)}_{i}} \cdot
        \frac{\partial A^{(L_0)}_{i}}{\partial Z^{(L_0)}_i}
    \]
    and
    \[
        W^{(L_0)}_{k, j} = \frac{\partial Z^{(L_0)}_i}{\partial A^{(L_0 - 1)}_{i}} 
    \]
    \[
        \Rightarrow \sum_{k = 1}^{m_{L_0}} \left( 
            \frac{\partial C}{\partial A^{(L_0)}_k} \cdot
            \frac{\partial A^{(L_0)}_k}{\partial Z^{(L_0)}_k} \cdot
            \frac{\partial Z^{(L_0)}_k}{\partial A^{(L_0 - 1)}_i}
        \right) = 
        \sum_{k = 1}^{m_{L_0}}\left(
            \delta^{(L_0)}_k \cdot W^{(L_0)}_{k, i}
        \right)
    \]
    \\
    We have 
    \[
        f_{L_0 - 1}^{\prime}\left(Z^{(L_0 - 1)}_i\right) =
        \frac{\partial A^{(L_0 - 1)}_i}{\partial Z^{(L_0 - 1)}_i}
    \]
    and 
    \[
        A^{(L_0-2)}_j = \frac{\partial Z^{(L_0 - 1)}_i}{\partial W^{(L_0 - 1)}_{i, j}}
    \]
    Let 
    \[
        \delta^{(L_0 - 1)}_i = \sum_{k = 1}^{m_{L_0}}\left(
            \delta^{(L_0)}_k \cdot W^{(L_0)}_{k, i}
        \right) \cdot f_{L_0 - 1}^{\prime}\left(Z^{(L_0 - 1)}_i\right)
    \]
    Therefore :
    \[
        \frac{\partial C}{\partial W^{(L_0-1)}_{i, j}} = 
        \delta^{(L_0 - 1)}_i \cdot
        A^{(L_0-2)}_j
    \]
    We have the partial derivative of $C$ respect to $B^{(L_0 - 1)}_{i}$ :
    \[
        \frac{\partial C}{\partial B^{(L_0-1)}_{i}} = 
        \sum_{k = 1}^{m_{L_0}} \left( 
            \frac{\partial C}{\partial A^{(L_0)}_k} \cdot
            \frac{\partial A^{(L_0)}_k}{\partial Z^{(L_0)}_k} \cdot
            \frac{\partial Z^{(L_0)}_k}{\partial A^{(L_0 - 1)}_i}
        \right) \cdot
        \left(
            \frac{\partial A^{(L_0 - 1)}_i}{\partial Z^{(L_0 - 1)}_i} \cdot
             \frac{\partial Z^{(L_0 - 1)}_i}{\partial B^{(L_0 - 1)}_{i}}
        \right) = \delta^{(L_0 - 1)}_i
    \]  
    \section*{Gradient of layer L $(1 \leq L \leq L_0)$ :}
    Similarly to the above layer, we obtain :

    \[
        \delta^{(L)}_i = 
        \begin{cases}
            \left(A^{(L_0)}_i - Y_i\right) \cdot f^{\prime}_{L_0}\left(Z^{(L_0)}_i\right), \ L = L_0\\
            \left(
                \sum_{k = 1}^{m_{L+1}}\left(
                    \delta^{(L + 1)}_k \cdot W^{(L + 1)}_{k, i}
                \right) 
            \right) \cdot f_{L}^{\prime}\left(Z^{(L)}_i\right), \ 1 \leq L < L_0
        \end{cases}
    \]
    and
    \[
        \frac{\partial C}{\partial W^{(L)}_{i, j}} = 
        \delta^{(L)}_i \cdot
        A^{(L-1)}_j
    \]
    \[
        \frac{\partial C}{\partial B^{(L)}_{i}} = 
        \delta^{(L)}_i
    \]
    \section*{Matrix Form :}
    We denote some matrices as follows:
    \\
    Activation matrix :
    \[
        A^{(L)} = 
        \begin{bmatrix}
            A^{(L)}_1  \\ A^{(L)}_2 \\ \hdots \\ A^{(L)}_{n_L} 
        \end{bmatrix} \ (n_L \times 1)
    \]
    Weight matrix : 
    \[
        W^{(L)} = 
        \begin{bmatrix}
            W^{(L)}_{1, 1} & W^{(L)}_{1, 2} & \hdots & W^{(L)}_{1, n_L} \\
            W^{(L)}_{2, 1} & W^{(L)}_{2, 2} & \hdots & W^{(L)}_{2, n_L} \\
            \hdots & \hdots & \hdots & \hdots \\
            W^{(L)}_{m_L, 1} & W^{(L)}_{m_L, 2} & \hdots & W^{(L)}_{m_L, n_L} 
        \end{bmatrix} \ (m_L \times n_L)
    \]
    Bias matrix : 
    \[
        B^{(L)} = 
        \begin{bmatrix}
            B^{(L)}_1  \\ B^{(L)}_2 \\ \hdots \\ B^{(L)}_{m_L} 
        \end{bmatrix} \ (m_L \times 1)
    \]
    From these matrices, we derive :
    \[
        Z^{(L)} = W^{(L)} \times A^{(L - 1)} + B^{(L)} = 
        \begin{bmatrix}
            Z^{(L)}_1  \\ Z^{(L)}_2 \\ \hdots \\ Z^{(L)}_{m_L}
        \end{bmatrix} \ (m_L \times 1)
    \]
    We calculate $A^{(L)}$ from $Z^{(L)}$ as follows:
    \[
        A^{(L)} = f_L\left(Z^{(L)}\right) = 
        \begin{bmatrix}
            f_L\left(Z^{(L)}_1\right)  \\ 
            f_L\left(Z^{(L)}_2\right) \\ 
            \hdots \\ 
            f_L\left(Z^{(L)}_{m_L}\right)
        \end{bmatrix} \ (m_L \times 1)
    \]
    Label matrix:
    \[
        Y = 
        \begin{bmatrix}
            Y_1  \\ Y_2 \\ \hdots \\ Y_{m_{L_0}} 
        \end{bmatrix} \ (m_{L_0} \times 1)
    \]
    Let consider $\delta^{(L)}_i$ :
    \\

    - With $L = L_0$ :
    \\
    We have :
        \[
            A^{(L_0)} - Y = 
            \begin{bmatrix}
                A^{(L_0)}_1 - Y_i  \\ 
                A^{(L_0)}_2 - Y_i \\ 
                \hdots \\ 
                A^{(L_0)}_{m_{L_0}} - Y_{m_{L_0}} 
            \end{bmatrix} \ (m_{L_0} \times 1)
        \]
        \[
            f^{\prime}_L\left(Z^{(L_0)}\right) = 
            \begin{bmatrix}
                f_L\left(Z^{(L_0)}_1\right)  \\ 
                f_L\left(Z^{(L_0)}_2\right) \\ 
                \hdots \\ 
                f_L\left(Z^{(L_0)}_{m_{L_0}}\right)
            \end{bmatrix} \ (m_{L_0} \times 1)
        \]
    From the formula of $\delta$ above, we obtain :
        \[
            \delta^{(L_0)} = \left(A^{(L_0)} - Y\right) \odot f^{\prime}_L\left(Z^{(L_0)}\right) 
        \]
        \[
            = \begin{bmatrix}
                \left(A^{(L_0)}_1 - Y_1\right) . f_L\left(Z^{(L_0)}_1\right)  \\ 
                \left(A^{(L_0)}_2 - Y_2\right) . f_L\left(Z^{(L_0)}_2\right) \\ 
                \hdots \\ 
                \left(A^{(L_0)}_{m_{L_0}} - Y_i\right) . f_L\left(Z^{(L_0)}_{m_{L_0}}\right)
            \end{bmatrix}
        \]
        \[
            = \begin{bmatrix}
                \delta^{(L_0)}_1  \\ 
                \delta^{(L_0)}_2 \\ 
                \hdots \\ 
                \delta^{(L_0)}_{m_{L_0}}
            \end{bmatrix} \ (m_{L_0} \times 1)
        \]
    - With  $ 1 \le L < L_0$ :
    \\
    We have :
        \[
            \left(W^{(L + 1)}\right)^T \times \delta^{(L + 1)} 
            =\begin{bmatrix}
                \sum_{k = 1}^{m_{L+1}}\left(
                    W^{(L + 1)}_{k, 1} \cdot \delta^{(L + 1)}_k  
                \right) \\
                \sum_{k = 1}^{m_{L+1}}\left(
                    W^{(L + 1)}_{k, 2} \cdot \delta^{(L + 1)}_k  
                \right) \\
                \hdots \\
                \sum_{k = 1}^{m_{L+1}}\left(
                    W^{(L + 1)}_{k, {n_{L+1}}} \cdot \delta^{(L + 1)}_k  
                \right) \\
            \end{bmatrix} (n_{L+1} \times 1)
        \]
        \[
            f^{\prime}_{L}\left(Z^{(L)}\right) = 
            \begin{bmatrix}
                f_L\left(Z^{(L)}_1\right)  \\ 
                f_L\left(Z^{(L)}_2\right) \\ 
                \hdots \\ 
                f_L\left(Z^{(L)}_{m_{L}}\right)
            \end{bmatrix} \ (m_{L} \times 1)
        \]
    Output of layer L ($A^{(L)}$) is the input of layer L + 1, thus $m_L = n_{L+1}$
    \\
    Based on the formula of $\delta$ that we metioned above :
    \[
        \left(
            \sum_{k = 1}^{m_{L+1}}\left(
                \delta^{(L + 1)}_k \cdot W^{(L + 1)}_{k, i}
            \right) 
        \right) \cdot f_{L}^{\prime}\left(Z^{(L)}_i\right), \ 1 \leq L < L_0
    \]
    We obtain $\delta$ matrix : 
    \[
        \delta^{(L)} = \left(\left(W^{(L + 1)}\right)^T \times \delta^{(L + 1)} \right) \odot
        f^{\prime}_{L}\left(Z^{(L)}\right)
    \]
    \[
        =\begin{bmatrix}
            \sum_{k = 1}^{m_{L+1}}\left(
                W^{(L + 1)}_{k, 1} \cdot \delta^{(L + 1)}_k 
            \right) \cdot  f_L\left(Z^{(L)}_1\right) \\
            \sum_{k = 1}^{m_{L+1}}\left(
                W^{(L + 1)}_{k, 2} \cdot \delta^{(L + 1)}_k  
            \right) \cdot f_L\left(Z^{(L)}_2\right) \\
            \hdots \\
            \sum_{k = 1}^{m_{L+1}}\left(
                W^{(L + 1)}_{k, {n_{L+1}}} \cdot \delta^{(L + 1)}_k  
            \right) \cdot f_L\left(Z^{(L)}_{m_{L}}\right) 
        \end{bmatrix} (m_{L} \times 1)
    \]
    \section*{Conclusion :}
    Activation matrix :

    Weight matrix :

    Bias matrix :

    Z matrix :

    Delta matrix :

    \section*{Layer L ($1 \leq L \leq L_0$): }
    Input :
    \[
        A^{(L-1)} \ \ \ \ (n_L \times 1)
    \]
    Weight : 
    \[
        W^{(L)} \ \ \ \ (m_L \times n_L) 
    \]
    Bias : 
    \[
        B^{(L)} \ \ \ \ (m_L \times 1)
    \]
    \[
        z^{(L)}_i = \sum_{j = 1}^{n_L}w^{(L)}_{i, j}a^{(L-1)}_j + b^{(L)}_j
        \Rightarrow Z^{(L)} = W^{(L)} \cdot A^{(L-1)} + B^{(L)} \ \ \ \ (m_L \times 1)
    \]
    Class weight:

    We choose the activation functions as follows :
    \[
        f_L(z) = \begin{cases}
            ReLu(z), 1 \leq L \le L_0 \\
            \sigma(z), L = L_0
        \end{cases}
    \]
    + ReLU function : 
    \[
        ReLU(z) = \begin{cases}
            z, z > 0 \\
            0, z \leq 0
        \end{cases}
    \]
    + Softmax function : 
    \[
        \sigma(z_i) = \sigma_i = \frac{e^{z_i}}{\sum_{j}e^{z_j}}
    \]
    Output : 
    \[ 
        a^{(L)}_i = f_L(z^{(L)}_i)
        \Rightarrow A^{(L)} = f_L(Z^{(L)}) \ \ \ \ (m_L \times 1)
    \] 
    \section*{Cross Entroy Loss:}
    Class weight : 
    \[
        \Omega = [\omega_i] \ \ \ \ (m_{L_0} \times 1)
    \]
    Label : 
    \[
        Y \ \ \ \ (m_{L_0} \times 1)
    \]
    Loss function :
    \[
        C = -\sum_{i = 1}^{m_{L_0}}\omega_iy_i\ln\left(
            a^{(L_0)}_i
        \right)
    \]

    \section*{Gradient Layer $L_0$ :}
    We have the partial derivative of $C$ respect to $W^{(L_0)}_{i, j}$ : 
    \[
        \frac{\partial C}{\partial w^{(L_0)}_{i, j}} = 
        \sum_{k = 1}^{m_{L_0}}\left(
            \frac{\partial C}{\partial a^{(L_0)}_{k}} \cdot
            \frac{\partial a^{(L_0)}_{k}}{\partial z^{(L_0)}_i} 
        \right) \cdot
        \frac{\partial z^{(L_0)}_i}{\partial w^{(L_0)}_{i, j}}
    \]  
    \[
        \frac{\partial C}{\partial a^{(L_0)}_{k}} = 
        \frac{-y_k\omega_k}{a_k^{(L_0)}}
    \]
    \[
        \frac{\partial a^{(L_0)}_{k}}{\partial z^{(L_0)}_i} = 
        \frac{\partial \sigma_k}{\partial z^{(L_0)}_i} = \begin{cases}
            a^{(L_0)}_{i}(1-a^{(L_0)}_{i}) , k = i \\
            -a^{(L_0)}_{i}a^{(L_0)}_{k} , k \neq i
        \end{cases}
    \]
    Let
    \[
        \delta^{(L_0)}_i = 
        \sum_{k = 1}^{m_{L_0}}\left(
            \frac{\partial C}{\partial a^{(L_0)}_{k}} \cdot
            \frac{\partial a^{(L_0)}_{k}}{\partial z^{(L_0)}_i}
        \right) = 
        \frac{-y_i\omega_i}{a_i^{(L_0)}} \cdot a^{(L_0)}_{i}(1-a^{(L_0)}_{i}) + 
        \sum_{k \neq i}\left(
            \frac{y_k\omega_k}{a_k^{(L_0)}}a^{(L_0)}_{i}a^{(L_0)}_{k}
        \right)
    \]
    \[
        = -y_i\omega_i + \sum_{k = 1}^{m_{L_0}}\left(
            \frac{y_k\omega_k}{a_k^{(L_0)}}a^{(L_0)}_{i}a^{(L_0)}_{k}
        \right) = 
        -y_i\omega_i + \sum_{k = 1}^{m_{L_0}}y_k\omega_ka^{(L_0)}_{i} = 
        \left(a^{(L_0)}_{i}\sum_{k = 1}^{m_{L_0}}y_k\omega_k\right) - y_i\omega_i
    \]
    There is only one $y_k = 1$ in the label, thus:
    \[
        \sum_{k = 1}^{m_{L_0}}y_k = y_c = 1 
        \Rightarrow a^{(L_0)}_{i}\sum_{k = 1}^{m_{L_0}}y_k\omega_k - y_i\omega_i = a_i^{(L_0)}\omega_c - y_i\omega_i
    \] 
    Therefore : 
    \[
        \delta^{(L_0)}_i = a_i^{(L_0)}\omega_c - y_i\omega_i
    \]
    And 
    \[
        \frac{\partial z^{(L_0)}_i}{\partial w^{(L_0)}_{i, j}} = a^{(L_0-1)}_j
    \]  

    \[
        \Rightarrow \frac{\partial C}{\partial w^{(L_0)}_{i, j}} = 
        \delta^{(L_0)}_i \cdot a^{(L_0-1)}_j
    \]
