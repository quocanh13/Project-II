    \[

    \[
        y_n = F_n(x_n) + x_n = F_n(y_{n-1}) + y_{n-1}
    \]
    Therefore:
    \[
        \frac{\partial y_n}{\partial x_1} = 
        \frac{\partial F_n(y_{n-1})}{\partial x_1} + \frac{\partial y_{n-1}}{\partial x_1} =
        \frac{\partial F_n(y_{n-1})}{\partial y_{n-1}}\frac{\partial y_{n-1}}{\partial x_1} + \frac{\partial y_{n-1}}{\partial x_1} 
    \]
    \[
        =\left(\frac{\partial F_n(y_{n-1})}{\partial y_{n-1}} + 1\right) 
        \frac{\partial y_{n-1}}{\partial x_1} 
        =\left(\frac{\partial F_n(y_{n-1})}{\partial y_{n-1}} + 1\right) 
        \left(\frac{\partial F_{n-1}(y_{n-2})}{\partial y_{n-2}} + 1\right)
        \frac{\partial y_{n-2}}{\partial x_1} 
    \]
    \[
        =\prod_{i=1}^{n}\left(
            \frac{\partial F_i(y_{i-1})}{\partial y_{i-1}} + 1
        \right)\frac{\partial y_{0}}{\partial x_1} 
        = \prod_{i=1}^{n}\left(
            \frac{\partial F_i(y_{i-1})}{\partial y_{i-1}} + 1
        \right)\frac{\partial x_1}{\partial x_1} 
        = \prod_{i=1}^{n}\left(
            \frac{\partial F_i(y_{i-1})}{\partial y_{i-1}} + 1
        \right)
    \]
